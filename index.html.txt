        // Donn√©es des cr√©neaux (simples pour cet exemple)
        const bookedSlots = {
            '2025-11-28': ['10:00-11:00', '14:00-15:00'],
            '2025-11-29': ['11:00-12:00', '15:00-16:00']
        };

        const allTimeSlots = [
            '09:00-10:00',
            '10:00-11:00',
            '11:00-12:00',
            '13:00-14:00',
            '14:00-15:00',
            '15:00-16:00',
            '16:00-17:00'
        ];

        let selectedTime = null;

        // Initialiser la date minimale √† aujourd'hui
        const dateInput = document.getElementById('date');
        const today = new Date().toISOString().split('T')[0];
        dateInput.min = today;

        // G√©n√©rer les cr√©neaux horaires
        function generateTimeSlots() {
            const selectedDate = document.getElementById('date').value;
            const container = document.getElementById('timeSlots');
            container.innerHTML = '';

            if (!selectedDate) {
                container.innerHTML = '<p style="grid-column: 1/-1; color: #999; text-align: center;">S√©lectionnez une date</p>';
                return;
            }

            const booked = bookedSlots[selectedDate] || [];

            allTimeSlots.forEach(slot => {
                const button = document.createElement('button');
                button.type = 'button';
                button.className = 'time-slot';
                button.textContent = slot;

                if (booked.includes(slot)) {
                    button.classList.add('booked');
                    button.disabled = true;
                } else {
                    button.classList.add('available');
                    button.addEventListener('click', (e) => {
                        e.preventDefault();
                        selectTimeSlot(button, slot);
                    });
                }

                container.appendChild(button);
            });
        }

        function selectTimeSlot(button, time) {
            // Retirer la s√©lection pr√©c√©dente
            document.querySelectorAll('.time-slot.selected').forEach(slot => {
                slot.classList.remove('selected');
            });

            // Ajouter la nouvelle s√©lection
            button.classList.add('selected');
            selectedTime = time;
        }

        // √âv√©nement de changement de date
        document.getElementById('date').addEventListener('change', generateTimeSlots);

        // Soumission du formulaire
        document.getElementById('bookingForm').addEventListener('submit', (e) => {
            e.preventDefault();

            const name = document.getElementById('name').value;
            const email = document.getElementById('email').value;

            document.getElementById('confirmTime').textContent = selectedTime;
